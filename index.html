Precise

Nanosecond timer for node.js

API

start()

Starts a timer

stop()

Stops a timer

diff()

Returns the nanoseconds between start() & stop()

Example

var precise = require('precise'),
    timer   = precise().start();

setTimeout(function () {
    console.log(timer.stop().diff());
}, 1000);
